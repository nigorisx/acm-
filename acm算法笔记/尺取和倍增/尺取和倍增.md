## 尺取（双指针）

·***双指针的复杂度为O（n）***可以用线性的复杂度维护区间，例如维护滑动窗口

```c++
//滑动窗口模板1
#include <iostream>
#include <deque>
#include <vector>
using namespace std;
struct el
{
    int v;
    int id;
};
int n,m;
vector<el> list;
void fmin()
{
    deque<el>q;
    for(int i=0;i<n;i++)
    {
        while(!q.empty() && i - q.front().id >= m)
        {
            q.pop_front();
        }
        while(!q.empty()&&q.back().v>list[i].v)
        {
            q.pop_back();
        }
        q.push_back(list[i]);
        if(i>=m-1)
        {
            cout<<q.front().v<<" ";
        }
    }
    cout<<endl;
}
void fmax()
{
    deque<el>q;
    for(int i=0;i<n;i++)
    {
        while(!q.empty() && i - q.front().id >= m)
        {
            q.pop_front();
        }
        while(!q.empty()&&q.back().v<list[i].v)
        {
            q.pop_back();
        }
        q.push_back(list[i]);
        if(i>=m-1)
        {
            cout<<q.front().v<<" ";
        }
    }
    cout<<endl;
}
int main()
{
    cin>>n>>m;
    list.resize(n);
    for(int i=0;i<n;i++)
    {
        int x;
        cin>>x;
        list[i]={x,i};
    }//

     fmin();
     fmax();
}
```

```c++
//滑动窗口模板2
#include <bits/stdc++.h>
using namespace std;
#define int long long

// 一维数组滑动窗口模板
int slidingWindow(vector<int>& arr, int k) {
    int left = 1, right = 1;      // 下标从1开始
    int sum = 0, ans = 0;
    int n = arr.size() - 1;       // 有效长度
    
    while (right <= n) {
        // 1. 右指针移动，扩展窗口
        sum += arr[right];
        right++;
        // 2. 条件不满足时，移动左指针收缩窗口
        //条件可以是窗口大小也可以是窗口内元素值的总和，比较灵活
        while (sum > k && left < right) {
            sum -= arr[left];
            left++;
        }
        // 3. 更新答案
        if (sum <= k) {
            ans += right - left;  // 统计子数组个数
        }
    }
    return ans;
}
//定一移一
```

## 题目的一些启发

You are given an array $a$ of $n$ integers, and $q$ queries. Each query is represented by two integers $l$ and $r$ ($1 \le l \le r \le n$). Your task is to find, for each query, two indices $i$ and $j$ (or determine that they do not exist) such that:

- $l \le i \le r$;

- $l \le j \le r$;

- $a_i \ne a_j$.

In other words, for each query, you need to find a pair of different elements among $a_l, a_{l+1}, \dots, a_r$, or report that sucha pair does not exist.

```c++
可以通过传递性，将每个相等的数都指向最靠近的不相同的数。那么假若区间两侧指向的数不同，则证明区间中肯定不止一种数。也许左右边界的数会相同，但右边界指向的数和它左侧的相邻元素必定都在[L,R]内，用q[r]表示r指向的数，q[r-1]!=q[r]必定存在，那么满足条件的i，j就找到的。类似于并查集中等号和不等号那道题，等号可以传递，不等号会打断传递，那么信息就需要更新，以不同作为更新的依据，又应为索引升序，即推得上方结论。
从链式的角度思考，相等的元素在同一传递链之中，每个节点指向头节点，每出现不等号，链被截断，要开新链，头节点也要更新
```

```c++
#include <bits/stdc++.h>
using namespace std;
void solve()
{
    int n;
    cin>>n;
    vector<int>v(n+1);
    vector<int>q(n+1);
    for(int i=1;i<=n;i++)
    {
        cin>>v[i];
    }
    for(int i=2;i<=n;i++)
    {
        if(v[i]==v[i-1])
        q[i]=q[i-1];//传递头节点信息
        else
        {
            q[i]=i-1;//更新头节点
        }
    }//O（n）
    int t1;
    cin>>t1;
    for(int i=0;i<t1;i++)//0(1)
    {
        int l,r;
        cin>>l>>r;//输入左右节点
        if(q[l]==q[r])
        {
            cout<< -1<<" "<< -1 <<endl;//仍在同一传递链中
        }
        else{
            cout<<q[r]<<" "<<q[r]+1<<endl;//不在同一传递链，返回上一条链的尾部和此时链的头部，满足条件
        }
    }
    cout<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        solve();
    }
}
```

***There's a new game Monocarp wants to play. The game uses a deck of $n$ cards, where the $i$\-th card has exactly one integer $a_i$ written on it. At the beginning of the game, on the first turn, Monocarp can take any card from the deck. During each subsequent turn, Monocarp can take exactly one card that has either the same number as on the card taken on the previous turn or a number that is one greater than the number on the card taken on the previous turn. In other words, if on the previous turn Monocarp took a card with the number $x$, then on the current turn he can take either a card with the number $x$ or a card with the number $x + 1$. Monocarp can take any card which meets that condition, regardless of its position in the deck. After Monocarp takes a card on the current turn, it is removed from the deck. According to the rules of the game, the number of distinct numbers written on the cards that Monocarp has taken must not exceed $k$. If, after a turn, Monocarp cannot take a card without violating the described rules, the game ends. Your task is to determine the maximum number of cards that Monocarp can take from the deck during the game, given that on the first turn he can take any card from the deck.***

```c++
为了维护滑动窗口，我们会将要维护的变量单独提出来放入数组。数组中的每个元素都是窗口中独立的元素。由于每个元素的数目，也可以说是权值，在维护时并不会对窗口本身造成影响，那么我们只要想办法可以从元素找到权值即可，map就是不错的选择。再额外用maxn去更新最值，那么在维护的滑动窗口合法的前提下即可维护到的最大值一定是符合题意的
```

```c++
#include <bits/stdc++.h>
using namespace std;
#define int long long
void solve()
{
    int n, k;
    cin >> n >> k;
    vector<int> v(n);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    map<int, int> list;
    for (int i = 0; i < n; i++) {
        list[v[i]]++;//提供直接查询权值的结构
    }
    vector<int> nums;
    for (auto p : list) {
        nums.push_back(p.first);//分离数字和权值
    }
    sort(nums.begin(), nums.end());//升序维护
    int  sum = 0;
    int maxn = -1;
    int l = 0;
    int r = nums.size();
    for (int r1 = 0; r1 < r; r1++) {
        if (r1 > 0 && nums[r1] != nums[r1 - 1] + 1) {//不满足窗口要求，清空滑动窗口，创建新窗口
            sum = 0;
            l = r1;
        }
        sum += list[nums[r1]];
        while (nums[r1] - nums[l] + 1 > k) {//维护窗口元素数目<=k
            sum -= list[nums[l]];//更新掉所有去除元素的相关变量
            l++;//收束窗口
        }
        maxn = max(maxn, sum);//更新最值
    }
    cout << maxn << endl;
}
signed main()
{
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
