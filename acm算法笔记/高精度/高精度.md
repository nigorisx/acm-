## 方法二

### 高精度算法——数组法

- 1. 核心思想：将参与运算的数的每一位用数组存起来，然后根据（多余删除）按照小学列竖式的计算方式按位计算。
- 1. 适用场景：超过int128的存储范围（十进制下超过37位）
- 3.优点：覆盖的数据范围更广，乘法速度更快（相比Python）
- 1. 缺点：代码量大，浪费赛场时间，容易出错。

void Erase0(vector<int>& a) {
    int R = a.size() - 1;
    while (a[R] == 0 && R > 0) {
        R--;
    }
    a.erase(a.begin() + R + 1, a.end());
}
void Normalize(vector<int> &a) {
    Erase0(a);
    int n = a.size();
    if (n == 0) return;
    bool f = 1;
    if (a[n - 1] < 0) {
        f = 0;
        for (int i = 0; i < n; i++) {
            a[i] = -a[i];
        }
    }
    for (int i = 0; i < n - 1; i++) {
        if (a[i] < 0) {
            a[i] += 10;
            a[i + 1]--;
        }
    }
    if (f == 0) {
        for (int i = 0; i < n; i++) {
            a[i] = -a[i];
        }
    }
    Erase0(a);
}

		if (b % 2) {
			res = 1LL * res * a % P;
		}

vector<int> rev, roots{ 0, 1 };
void dft(vector<int>& a) {
	int n = a.size();
	if (int(rev.size()) != n) {
		int k = ctz(n) - 1;
		rev.resize(n);
		for (int i = 0; i < n; i++) {
			rev[i] = rev[i >> 1] >> 1 | (i & 1) << k;
		}
	}
	for (int i = 0; i < n; i++) {
		if (rev[i] < i) {
			swap(a[i], a[rev[i]]);
		}
	}
	if (roots.size() < n) {
		int k = ctz(roots.size());
		roots.resize(n);
		while ((1 << k) < n) {
			int e = power(31, 1 << (ctz(P - 1) - k - 1));
			for (int i = 1 << (k - 1); i < (1 << k); i++) {
				roots[2 * i] = roots[i]; roots[2 * i + 1] = 1LL * roots[i] * e % P;
			}
			k++;
		}
	}
	for (int k = 1; k < n; k *= 2) {
		for (int i = 0; i < n; i += 2 * k) {
			for (int j = 0; j < k; j++) {
				int u = a[i + j];
				int v = 1LL * a[i + j + k] * roots[k + j] % P;
				a[i + j] = (u + v) % P;
				a[i + j + k] = (u - v) % P;
			}
		}
	}
}
void idft(vector<int>& a) {
	int n = a.size();
	reverse(a.begin() + 1, a.end()); dft(a);
	int inv = (1 - P) / n;
	for (int i = 0; i < n; i++) {
		a[i] = 1LL * a[i] * inv % P;
	}
}
vector<int> mul(vector<int> a, vector<int> b) {
	int n = 1, tot = a.size() + b.size() - 1;
	while (n < tot) {
		n *= 2;
	}
	a.resize(n);
	b.resize(n);
	dft(a);
	dft(b);
	for (int i = 0; i < n; i++) {
		a[i] = 1LL * a[i] * b[i] % P;
	}
	idft(a);
	for (int i = 0; i < n; i++) {
		a[i] = (a[i] + P) % P;
	}
	a.resize(tot);
	return a;
}

		for (int i = 0; i < len; i++){if (a[i] >= 10){a[i+1] += a[i] / 10; a[i] %= 10;}}
		while(len > 1 and !a[len-1]){len--;} a.len = len;
		a.is_fushu = ((a == 0 or a.is_fushu == b.is_fushu) ? 0 : 1);
	}
	void operator /= (Int &a , Int b){a = a / b;} // O(n^2)
	void operator %= (Int &a , Int b){a = a % b;} // O(n^2)
}
