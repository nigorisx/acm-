## 最小生成树

### **prim算法求最小生成树**

**code**

```c++

#include <iostream>
#include <vector>
#include <queue>
#include <cstring>
using namespace std;
const int MAXN = 1000, INF = 0x3f3f3f3f;
// 邻接表存储图
vector<pair<int, int>> g[MAXN];
int dist[MAXN], n, m, res;
bool visited[MAXN];
// 堆优化 Prim 算法
void prim_heap() {
    // 初始化
    memset(dist, 0x3f, sizeof(dist));
    memset(visited, false, sizeof(visited));
    res = 0;
    // 小顶堆，存储 {距离, 顶点}
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    // 从顶点1开始
    dist[1] = 0;
    pq.push({0, 1});
    int cnt = 0; // 记录已加入最小生成树的顶点数
    while (!pq.empty() && cnt < n) {
        // 取出距离最小的顶点
        auto [d, u] = pq.top();
        pq.pop();
        // 如果这个顶点已经被访问过，跳过
        if (visited[u]) continue;
        // 标记为已访问，加入最小生成树
        visited[u] = true;
        res += d;
        cnt++;
        // 遍历所有邻接顶点
        for (auto &[v, w] : g[u]) {
            if (!visited[v] && w < dist[v]) {
                dist[v] = w;
                pq.push({w, v});
            }
        }
    }
    // 如果生成树包含的顶点数不足n个，说明图不连通
    if (cnt < n) {
        res = INF;
    }
}
int main() {
    cin >> n >> m;
    // 初始化邻接表
    for (int i = 1; i <= n; i++) {
        g[i].clear();
    }
    // 读入边
    for (int i = 1; i <= m; i++) {
        int a, b, w;
        cin >> a >> b >> w;
        g[a].push_back({b, w});
        g[b].push_back({a, w});
    }
    prim_heap();
    if (res == INF)
        cout << "orz";
    else
        cout << res;
    return 0;
}

```

### 性质及构造应用
- 双重性质
	- 1. ⼆叉搜索树性质中序遍历等于原数组序列，节点的左右相对位置与原序列⼀致。
    - 2. 堆性质每个节点的权值⼤于其⼦节点的权值。
- 应用场景:处理区间最值问题（ $RMQ$ ） , 将数组区间转换为树上的 $LCA$( 最近公共祖先问题 )
- 建树⽅法⽤单调栈维护⼩根堆，实现线性建树，复杂度为 $O(N)$

code
```c++
//建树模板
#include <bits/stdc++.h>
using namespace std;
#define int long long 
struct node
{
    int id;
    int val;
};
signed main()
{
    ios::sync_with_stdio(0);
    cout.tie(0);cin.tie(0);
    int n;
    cin>>n;
    vector<int>v(n+1);
    vector<int>l(n+1);
    vector<int>r(n+1);
    deque<node>dq;
    for(int i=1;i<=n;i++)
    {
        int val;cin>>val;
        int last=-1;
        while(dq.size()>0&&val<dq.front().val)
        {
            last=dq.front().id;
            dq.pop_front();
        }
        if(last!=-1)l[i]=last;
        if(dq.size()>0)
        {
            auto top=dq.front();
            r[top.id]=i;   
        }
        dq.push_front({i,val});
    }
    int ans1=0,ans2=0;
    for(int i=1;i<=n;i++)
    {
        ans1^=(i*(l[i]+1));
        ans2^=(i*(r[i]+1));
    }
    cout<<ans1<<" "<<ans2;
}
```

## 树形DP

### **树形DP的基本特征**：
- 状态常定义为$dp_{u,\dots}$,表示以u为根节点的最优解。
- 转移顺序通过$DFS$ 递归处理。先递归子节点，回溯时更新父节点。
### 常见例题
$eg1.$ ***树的直径问题*** 

   **code示例**

```c++
#include <bits/stdc++.h>
using namespace std;
int dis[1000005];
vector< vector<int> >g;
vector<int>d1;vector<int>d2;
int d=0;
void dfs(int u, int fa) {
  d1[u] = d2[u] = 0;
  for (int v : g[u]) {
    if (v == fa) continue;
    dfs(v, u);
    int t = d1[v] + 1;
    if (t > d1[u])
      d2[u] = d1[u], d1[u] = t;
    else if (t > d2[u])
      d2[u] = t;
  }
  d = max(d, d1[u] + d2[u]);
}
int main()
{
    int n;
    cin>>n;
    g.resize(n+1);
    d1.resize(n+1);
    d2.resize(n+1);
    for(int i=1;i<=n-1;i++)
    {
        int from,to;
        cin>>from>>to;
        g[from].push_back(to);
        g[to].push_back(from);
    }
    dfs(1,0);
    cout<<d;
}
```

 eg2.***最大权独立集***
  [**没有上司的舞会**](https://www.luogu.com.cn/problem/P1352) 有 $n$ 名职员，编号为 $1 \sim n$  ，他们的关系就像一棵以老板为根的树，父节点就是子节点的直接上司。每个职员有一个快乐指数，用整数$H_i$给出，现在要召开一场周年庆宴会，不过，没有职员愿意和直接上司一起参会。在满足这个条件的前提下，主办方希望邀请一部分职员参会，使得所有参会职员的快乐指数总和最大，求这个最大值。
  思路:对于每个职员，即每个节点，都有两种状态，参加和不参加，可以用二维数组的第二维度来记录(0/1 分别表示参加和不参加)，因为$\text{DFS}$ 是向上更新，所以假设**son**是u的子节点，有转移方程如下
$$
\left \{\begin{array}{ll}
dp[u][0]+=dp[son][1] \quad 当前节点不参加\ \\
dp[u][1]+=max(dp[son][0],dp[son][1])\quad当前节点参加，子节点状态待定\\
\end{array}\right.
$$




```c++
#include <bits/stdc++.h>
using namespace std;
int happy[10000];
int dp[10000][2];
vector<vector<int> >v;
int n;
void dfs(int u,int fa)
{
    dp[u][1]=happy[u];
    for(auto e:v[u])
    {
        if(e==fa)continue;
        dfs(e,u);
        dp[u][0]+=max(dp[e][0],dp[e][1]);
		dp[u][1]+=dp[e][0];
    }
}
int main()
{
    cin>>n;
    v.resize(n+1);
    for(int i=1;i<=n;i++)
    {
        cin>>happy[i];
    }
    for(int i=1;i<=n-1;i++)
    {
        int from,to;
        cin>>from>>to;
        v[from].push_back(to);
        v[to].push_back(from);
    }
    dfs(1,0);
    cout<<max(dp[1][1],dp[1][0]);
}

```

### 换根dp（特别的树形dp）

***树形 DP 中的换根 DP 问题又被称为二次扫描，通常不会指定根结点，并且根结点的变化会对一些值，例如子结点深度和、点权和等产生影响．***

***通常需要两次 DFS，第一次 DFS 预处理诸如深度，点权和之类的信息，在第二次 DFS 开始运行换根动态规划．***
