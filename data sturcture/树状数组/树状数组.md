##  树状数组

操作总览

```c++
#include <bits/stdc++.h>
using namespace std;

class BIT {
private:
    int n;                 // 数组大小
    vector<int> t;         // 树状数组
    
    int lowbit(int x) {    // 核心函数
        return x & -x;
    }

public:
    // 构造函数1：指定大小
    BIT(int _n) : n(_n), t(_n + 1, 0) {}
    
    // 构造函数2：从数组构建
    BIT(vector<int>& a) {
        n = a.size();
        t = vector<int>(n + 1, 0);
        for (int i = 1; i <= n; i++) {
            add(i, a[i - 1]);
        }
    }
    
    // 单点修改
    void add(int x, int k) {
        for (int i = x; i <= n; i += lowbit(i)) {
            t[i] += k;
        }
    }
    
    // 前缀和
    int sum(int x) {
        int res = 0;
        for (int i = x; i; i -= lowbit(i)) {
            res += t[i];
        }
        return res;
    }
    
    // 区间查询
    int query(int l, int r) {
        return sum(r) - sum(l - 1);
    }
};
```



```c++
#include <bits/stdc++.h>
using namespace std;

struct BIT {
    int n;
    vector<int> d;
    
    // 构造函数：接收从1开始的数组
    BIT(vector<int>& a) {
        n = a.size() - 1;  // a[0]是占位符，实际数据从a[1]开始
        d.resize(n + 2, 0);
        
        for (int i = 1; i <= n; i++) {
            int val = a[i];  // 直接使用a[i]，因为a[1]是第一个元素
            _add(i, val);
            _add(i + 1, -val);
        }
    }
    
    int lowbit(int x) { return x & -x; }
    
    void _add(int x, int k) {
        for (int i = x; i <= n; i += lowbit(i)) d[i] += k;
    }
    
    int _sum(int x) {
        int res = 0;
        for (int i = x; i; i -= lowbit(i)) res += d[i];
        return res;
    }
    
    void range_add(int l, int r, int k) {
        _add(l, k);
        _add(r + 1, -k);
    }
    
    int query(int x) { return _sum(x); }
    
    void print() {
        cout << "当前数组: [";
        for (int i = 1; i <= n; i++) {
            cout << query(i);
            if (i < n) cout << ", ";
        }
        cout << "]" << endl;
    }
};

int main() {
    // 创建从1开始的数组，a[0]是占位符
    vector<int> a = {0, 2, 5, 7, 1, 8};  // 实际数据从a[1]开始
    // 位置:     0  1  2  3  4  5
    // 值:       占位 2  5  7  1  8
    
    BIT bit(a);
    cout << "初始数组: ";
    bit.print();
    
    return 0;
}
```
