## 单源最短路

### spfa

**首先 spfa 是一个时间复杂度在 $O(VE)$ 的高复杂度算法，一般不会拿他跑最短路，但是它可以用来判断 ** 图中是否存在负权回路 **，因为如果存在负环，那么就不存在最短路。相当于 spfa 的 while 循环永远出不来。**

1. 方法 1：统计每个点入队的次数，如果某个点入队 n 次，则说明存在负环

2. 方法 2：统计当前每个点的最短路中所包含的边数，如果某点的最短路所包含的边数大于等于 n，则也说明存在环

   ```cpp
   #include <bits/stdc++.h>
   using namespace std;
   const int N=105;
   struct Edge {
       int to,nxt,w;
   }edge[N*N];
   int head[N],cnt;
   void add(int u,int v,int w) {
       edge[cnt].to=v;
       edge[cnt].nxt=head[u];
       edge[cnt].w=w;
       head[u]=cnt++;
   }
   int dis[N],vis[N];
   int n,m;
   const int INF=0x3f3f3f3f;
   void spfa() {
       for (int i=1;i<=n;i++)dis[i]=INF;
       queue<int>q;
       dis[1]=0;vis[1]=1;q.push(1);
       while (!q.empty()) {
           int u=q.front();q.pop();vis[u]=0;
           int v,w;
           for (int i=head[u];i!=-1;i=edge[i].nxt) {
               v=edge[i].to;
               w=dis[u]+edge[i].w;
               if (dis[v]>w) {
                   dis[v]=w;
                   if (!vis[v])vis[v]=1,q.push(v);
               }
           }
       }
   }
   void solve() {
       string s;
       cin>>n>>m>>s;
       for (int i=0;i<=n;i++) {
           head[i]=-1;
       }
       cnt=0;
    	//....
       //建图
       spfa();
       if (dis[n]==INF)cout<<"-1\n";
       else cout<<dis[n]<<"\n";
   }
   int main() {
       solve();
   }
   ```



## 多源最短路

### Floyd

**初始化，存初始的路径，再依次以 k = 1~n 为中间节点，遍历所有点对 (a, b) ，用 a-> k+k-> b 尝试更新 a-> b 的最短路径，即看从 a 到 k 再从 k 到 b 的路径是否比目前 a 到 b 存的最短路径短。**

复杂度 $O(n^3)$, 效率极其慢

```cpp
#include<bits/stdc++.h>
using namespace std;
#define int long long
namespace SOL
{
    using ll = long long;
    using pii = pair<int, int>;
    using vi = vector<int>;
    using vvi = vector<vector<int>>;
    const int INF = 0x3f3f3f3f;
    const ll LLINF = 0x3f3f3f3f3f3f3f3fLL;
    template<typename T, typename L>
    bool chkmax(T &x, L y) { if (x < y) return x = y, true; return false; }
    template<typename T, typename L>
    bool chkmin(T &x, L y) { if (y < x) return x = y, true; return false; }
    int vex[105][105];
    void solve()
    {
        int n,m;
        cin>>n>>m;
        memset(vex,0x3f3f3f3f,sizeof(vex));
        for (int i=1;i<=n;i++) {
            for (int j=1;j<=n;j++) {
                if (i==j)vex[i][j]=0;
            }
        }
        for (int i=1;i<=m;i++) {
            int u,v,w;
            cin>>u>>v>>w;
            vex[u][v]=min(w,vex[u][v]);//这里考虑了重边
            vex[v][u]=vex[u][v];
        }
        for (int k=1;k<=n;k++) {
            for (int i=1;i<=n;i++) {
                for (int j=1;j<=n;j++) {
                    vex[i][j]=min(vex[i][j],vex[i][k]+vex[k][j]);
                }
            }
        }
        for (int i=1;i<=n;i++) {
            for (int j=1;j<=n;j++) {
                cout<<vex[i][j]<<" ";
            }
            cout<<"\n";
        }
    }

}
signed main()
{
    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    int t;
    t=1;
    while(t--)
    {
        SOL::solve();
    }
}
```
